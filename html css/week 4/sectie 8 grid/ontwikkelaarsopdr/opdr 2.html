Container properties
//Creates a grid
display: grid || inline-grid
//Creates grid columns
Grid-template-columns: length || percentage || fr || line-name
//Creates grid rows
Grid-template-rows: length || percentage || fr || line-name
//Creates a grid with specified areas
grid-template-area: grid-area-name || none || .
//Creates rows, columns and areas in a single declaration
grid-template: none (initial value) || grid-template-rows / grid-template-columns
//Creates spaces between grid-items on the row axis
row-gap (grid-row-gap): length
//Creates spaces between grid-items on the column axis
column-gap (grid-column-gap): length
//Creates spaces between grid-items on the row and column axis
gap (grid-gap): grid-row-gap grid-column-rap
//Align grid items along the inline/row axis
Justify-items: start || end || center || stretch
//Align grid items along the block/column axis
Align-items: start || end || center || stretch
//Align grid items along the inline and block axis (shorthanded)
place-items: align-items || Justify-items
//Align the whole grid (container) along the inline/row axis
Justify-items: start || end || center || stretch || space-between || space-around || space-evenly
//Align the whole grid (container) along the block/column axis
Align-items: start || end || center || stretch | space-around || space-between || space-evenly
//Align the whole grid (container) along the inline and block axis 9shorthanded)
place-content: align-content || justify-content
//specifies the size of any auto-genrated grid row
grid-auto-rows: size (length) || percentage || fr
/Specifies the size of any auto-genrated grid column
grid-auto-column: size( (length) || percentage || fr
//Sets the algorithm that places the grid-items
grid-auto-flow: row || column || dense
//Shorthand for almost every property
grid: grid-template rows || grid-template-columns || grid-template-areas || grid-auto-rows || grid-auto-columns || grid-auto-flow
item properties
//Determinens the column start of an item within the grid
grid-column-start: line || span (number || name) || auto
//Determinens the row start of an item within the grid
grid-column-start: line || span (number || name) || auto
//Determinens the column end of an item within the grid
grid-column-start: line || span (number || name) || auto
//Determinens the row end of an item within the grid
grid-column-start: line || span (number || name) || auto
//Shorthand for column start and end
grid-column: grid-column-start || grid-column-end
//Shorthand for row start and end
grid-row: grid-row-start || grid-row-end
//Give a grid-area-name to a grid item
grid-area: name || row-start || column-start || row-end || column-end
//Aligns grid items inside a cell along the inline / row axis
justify-self: start || end || center || stretch
//Aligns grid items inside a cell along the block / column axis
Align-self: start || end || center || stretch
//Align a grid item with a cell along the row/inline and block/column axis
place-self: align-self || justify-self
